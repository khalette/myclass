    Base class for vehicles.
    
    def move(self):
        Base move method.
        raise NotImplementedError("Subclasses must implement this method.")


class Car(Vehicle):
    Car class inheriting from Vehicle.
    
    def move(self):
        Overrides the move method for Car.
